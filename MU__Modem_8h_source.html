69struct MU_Modem_Event
70{
71 ModemError error;
72 MU_Modem_Response type;
73 int32_t value;
74 const uint8_t *pPayload;
75 uint16_t payloadLen;
76 const uint8_t *pRouteNodes;
77 uint8_t numRouteNodes;
78
79 // --- Constructors ---
80 // 1. Default: Initialize everything to zero/null for safety
81 MU_Modem_Event() : error(ModemError::Ok), type(MU_Modem_Response::Idle), value(0), pPayload(nullptr), payloadLen(0), pRouteNodes(nullptr), numRouteNodes(0) {}
82
83 // 2. Helper for simple status events
84 MU_Modem_Event(ModemError err, MU_Modem_Response t)
85 : error(err), type(t), value(0), pPayload(nullptr), payloadLen(0), pRouteNodes(nullptr), numRouteNodes(0) {}
86
87 // 3. Helper for events with a value (RSSI, Channel, etc.)
88 MU_Modem_Event(ModemError err, MU_Modem_Response t, int32_t val)
89 : error(err), type(t), value(val), pPayload(nullptr), payloadLen(0), pRouteNodes(nullptr), numRouteNodes(0) {}
90};
